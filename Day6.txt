PROMPT 3 :
Make a sorting function which returns a new sorted vector and takes the original vector as input. This function will use the logic of merge sort.

RESULT :

vector<int> merge(const vector<int>& left, const vector<int>& right) {
    vector<int> result;
    result.reserve(left.size() + right.size());

    size_t i = 0, j = 0;
    while (i < left.size() && j < right.size()) {
        if (left[i] <= right[j]) {
            result.push_back(left[i++]);
        } else {
            result.push_back(right[j++]);
        }
    }
    while (i < left.size()) result.push_back(left[i++]);
    while (j < right.size()) result.push_back(right[j++]);

    return result;
}

vector<int> mergeSortHelper(vector<int> arr) {
    if (arr.size() <= 1) {
        return arr;
    }

    size_t mid = arr.size() / 2;
    vector<int> left(arr.begin(), arr.begin() + mid);
    vector<int> right(arr.begin() + mid, arr.end());

    left = mergeSortHelper(left);
    right = mergeSortHelper(right);

    return merge(left, right);
}

vector<int> mergeSort(const vector<int>& original) {
    return mergeSortHelper(original);
}
